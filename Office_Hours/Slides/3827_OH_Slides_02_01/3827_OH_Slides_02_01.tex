\section{Logistics}
\subsection{Course Grading}
\begin{frame}{Logistics: Course Grading}
    \begin{itemize}
        \item Raw score $S$ formula:
        \begin{gather*}
            S = 0.1H + \max{\left(0.3M + 0.6F, 0.45M + 0.45F\right)} 
        \end{gather*}
        where $H$ is homework, $M$ is midterm, and $F$ is final
    \end{itemize}
\end{frame}
